theta <- function(d, i) {
    agrupado = d %>% 
        slice(i) %>% 
        group_by(Project) %>% 
        summarise(media = Total_Ocorr/Commits, .groups = "drop")
        a = agrupado %>% pull(media)
        a
}

    d %>%
        ggplot(aes(
            xmin = low,
            x = mid,
            xmax = high,
            y = metodo
        )) +

        geom_point(color = "coral", size = 3) +
